        //  Program to print hollow star Rectangle

        import Input.*;
        class HollowRectangle
        {
            public static void main(String args[])
            {
                int row = Check.getDigit();

                for(int i = 0; i < row; i++)
                {
                    for(int j = 0; j < 2 * row; j++)
                    {
                        if(!(j == 0 || j == 2 * row - 1) && !(i == 0 || i == row - 1))
                            System.out.print("  ");
                        else
                            System.out.print("* ");
                    }
                    System.out.println();
                }
                        
            }
        }
